1. 문자 결합 연산자(+)
	문장을 여러개 결합
	
	역따옴표(` - 백틱) 
	
	템플릿 리터럴
	
	var a = "안녕하세요
		반갑습니다"; ---> 에러 발생
		
	var a = `안녕하세요
		반갑습니다`; --> 정상 인식
		
	
	템플릿 리터럴 - 보간 표현법
		변수의 값, 함수 호출을 문장에 그대로 포함이 가능한 방식
		
	문자 결합 연산자 방식
	var a = 1;
	var b = 2; 
	var c = a + " 더하기 " + b + "는 " + a + b;
	
	템플릿 리터럴 - 보간 표현법
	${변수명}
	var a = 1;
	var b = 2; 
	var c = `${a} 더하기 ${b} 는 ${a+b}`;
